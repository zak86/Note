метод split (розделітель), дозволяє перетворити рядок в масив, розбивши її по разделителю:

var names = 'Маша|Петя|Марина|Василий';

var arr = names.split('|', 2); // другий необов'язковий аргумент ограніченіє на кількість елементів у масиві

for (var i=0; i<arr.length; i++) { 

document.write(arr[i] + '<hr>');
}
