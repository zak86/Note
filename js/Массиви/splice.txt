<script type="text/javascript">
	var fruts = ["апельсин", "банан", "груша", "яблоко", "ананас"];

	fruts.splice(3, 2); //Видаляє декілька елементів масиву, 3-номер комірки, 2-кількість видаляємих комірок

	document.write(fruts + "<br />");

	fruts.splice(1, 2, "абрикос", "ківі"); //заміна комірок, 1-з якої потрібно видалити, 2-кількість видалених комірок, "абрикос", "ківі"-займають місце видалених комірок
	document.write(fruts);
